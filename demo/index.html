/* ───────────────────────────────────────────────────────────── INSPECTRA — AI Element Inspection Effect ───────────────────────────────────────────────────────────── */ /* ── 1. INSTALLATION ──────────────────────────────────────── */ // Option A — CDN (no build step required) <script src="https://cdn.jsdelivr.net/npm/inspectra/dist/inspectra.umd.min.js"></script> // After loading, the global object "Inspectra" is available. // All methods are also destructurable: const { inspect, uninspect, inspectAll } = Inspectra; // Option B — NPM // $ npm install inspectra import Inspectra from 'inspectra'; // default import (full object) import { inspect, uninspect, inspectAll } from 'inspectra'; // named imports /* ── 2. BASIC USAGE ───────────────────────────────────────── */ // Pass a CSS selector string… const ctrl = Inspectra.inspect('#my-element'); // …or a direct DOM reference const el = document.querySelector('.card'); const ctrl = Inspectra.inspect(el); // Remove the effect Inspectra.uninspect('#my-element'); Inspectra.uninspect(el); // Apply to every element matching a selector const ctrls = Inspectra.inspectAll('.ai-target'); // returns array of controls ctrls.forEach(c => c.destroy()); // remove all at once /* ── 3. CONTROL OBJECT ────────────────────────────────────── */ // inspect() returns a control object with three methods: ctrl.pause(); // freeze all animations (CSS + canvas particles) ctrl.resume(); // resume from where they stopped ctrl.destroy(); // remove the effect entirely and restore the original DOM /* ── 4. OPTIONS REFERENCE ─────────────────────────────────── */ // All options are optional. Defaults shown below. Inspectra.inspect('#el', { // ── Layout ────────────────────────────────────────────── block: false, // Set to true to force the wrapper div to display:block. // Useful for elements like <img>, <video>, block cards, etc. // Default: false (wrapper is inline-block) paused: false, // Start with all animations paused immediately. // Call ctrl.resume() to start them later. // ── Particles ─────────────────────────────────────────── particles: 4, // Number of floating light dots orbiting the element. // Range: 0 (none) to ~40 (dense). Default: 4. color: [130, 150, 255], // RGB array for particles, ring glow, and sweep tint. // Default: [130, 150, 255] (blue-violet) // Example: [80, 220, 180] for teal, [255, 120, 80] for orange // ── Global speed ──────────────────────────────────────── speed: 1, // Multiplier applied to ALL animation durations at once. // 1 = default speed // 2 = 2× faster (more intense / urgent feel) // 0.5 = half speed (slower, more ambient / calm) // 0.2 = very slow, almost meditative // Note: individual duration options below override speed. // ── Per-animation durations (override speed) ──────────── sweepDuration: '3.2s', // Light band that scrolls top→bottom inside the element. // Shorter = faster sweep. Default: '3.2s' ringDuration: '4s', // Rotating arc of light around the border. // Shorter = faster spin. Default: '4s' breatheDuration: '3.5s', // Outer glow pulse (inhale/exhale rhythm). // Longer = slower, more relaxed breathing. Default: '3.5s' holeDuration: '4s', // Edge vignette opacity cycle. // Default: '4s' }); /* ── 5. COMMON RECIPES ────────────────────────────────────── */ // Urgent / fast AI analysis feel Inspectra.inspect('#el', { speed: 2.5, particles: 12 }); // Calm ambient highlight Inspectra.inspect('#el', { speed: 0.4, particles: 3, color: [100, 180, 255] }); // Hover-to-inspect pattern el.addEventListener('mouseenter', () => Inspectra.inspect(el)); el.addEventListener('mouseleave', () => Inspectra.uninspect(el)); // Inspect, then auto-remove after 3 seconds const ctrl = Inspectra.inspect('#el'); setTimeout(() => ctrl.destroy(), 3000); // Block element (image, card, full-width div) Inspectra.inspect('#hero-image', { block: true, particles: 8 }); // Custom teal color, slow sweep, fast ring Inspectra.inspect('#el', { color: [60, 220, 180], sweepDuration: '6s', ringDuration: '1.5s', });
